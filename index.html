3.1. Kruskal's Algorithm

This algorithm follows an Edge-based Greedy approach:

1. Sort all edges of the graph in non-decreasing order of their weights.

2. Pick the smallest edge. If adding it doesn't form a cycle (checked via Union-Find), include it in the MST.

3. Repeat until there are \(n-1\) edges in the MST.

3.2. Prim's Algorithm

This algorithm employs a Vertex-based Greedy approach:

1. Start with an arbitrary vertex and include it in the MST set.

2. Find the minimum weight edge that connects a vertex in the MST to a vertex outside the MST.

3. Add the connected vertex to the MST set and repeat until all vertices are covered.

# Structure for managing sets (Cycle detection)
class DisjointSet:
    def __init__(self, n):
        self.parent = list(range(n))

    # Path compression
    def find(self, i):
        if self.parent[i] == i:
            return i
        self.parent[i] = self.find(self.parent[i])
        return self.parent[i]

    # Union by rank (simplified)
    def union(self, i, j):
        root_i = self.find(i)
        root_j = self.find(j)
        if root_i != root_j:
            self.parent[root_i] = root_j
            return True
        return False

def kruskal(n, edges):
    edges.sort() 
    dsu = DisjointSet(n)
    mst = []
    total_weight = 0
    for weight, u, v in edges:
        if dsu.union(u, v):
            mst.append((u, v, weight))
            total_weight += weight
            if len(mst) == n - 1: break
    return mst, total_weight

import heapq

def prim(n, adj):
    mst = []
    total_weight = 0
    visited = [False] * n
    pq = [(0, 0, -1)] # (weight, current, parent)
    
    while pq:
        weight, u, p = heapq.heappop(pq)
        if visited[u]: continue
        visited[u] = True
        total_weight += weight
        if p != -1: mst.append((p, u, weight))
        for v, w in adj[u]:
            if not visited[v]:
                heapq.heappush(pq, (w, v, u))
    return mst, total_weight
